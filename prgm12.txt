12) file permissions

#!/bin/bash
#set -x
if [ -f $1 ]
then
        if [ -r $1 ]
        then
                echo "$1 is having read permission"
        else
                echo "#1 is not having read permission"
        fi

        if [ -w $1 ]
        then
                echo "$1 is  having write permission"
        else
                echo "$1 is not gaving write permission"
        fi
        if [ -x $1 ]
        then
                echo "$1 is having execute permisson"
        else
                echo "$1 is not having execute permission"
        fi

        echo "$1 is file"
else
        echo "$1 is dir"
fi
